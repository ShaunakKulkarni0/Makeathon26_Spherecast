      - unit: str

      - defect_rate: dict | None
      - on_time_delivery: dict | None
      - years_in_business: int | None
      - audit_score: dict | None
  - moq: int
  - country_of_origin: str
  - incoterm: str
  - source_url: str | None

B) candidates (list[CrawledMaterial])
  - gleicher Feldaufbau wie original
  - mehrere Kandidatenobjekte in einer Liste

C) user_requirements (UserRequirements)

  - destination_country: str
